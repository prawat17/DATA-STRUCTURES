Q. Dynamic memory allocation [Program should include malloc, free, calloc, realloc]
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int *arr,i,n,m;
    printf("Input the values of the size\n");
    scanf("%d",&n);
    arr = (int*)malloc(n*sizeof(n));
    printf("Input the elements: \n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    free(arr);
    arr = (int*)calloc(n,sizeof(n));
    printf("Input the elements: \n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    printf("Enter the new size\n");
    scanf("%d",&m);
    arr = (int*)realloc(arr,m);
    printf("Input the elements: \n");
    for(i=0;i<m;i++){
        scanf("%d",&arr[i]);
    }
    free(arr);
}
